# General Loops

## For Loop

```javascript
for (let i = 0; i < 10; i++) {
  if (i === 2) {
    console.log("2 is my favorite number");
    continue; // When i = 2 print '2 is my favorite number' than 2
  }
  if (i === 5) {
    console.log("Stop the loop");
    break; // When i are equal to 5 stop the loop.
  }
  console.log("Number " + i);
}
```

## While Loop

```javascript
let i = 0;

while (i < 10) {
  console.log("Number " + i);
  i++;
}
```

## Do While

```javascript
let i = 100;

do {
  console.log("Number " + i);
  i++;
} while (i < 10); // In this case i = 100 so loop run one time cause i are bigger than 10
```

## Loop Through Array

```javascript
const cars = ["Ford", "Chevy", "Honda", "Toyota"];

for (let i = 0; i < cars.length; i++) {
  console.log(cars[i]);
}
```

## For Each

```javascript
cars.forEach(function (car, index, array) {
  console.log(`${index} : ${car}`);
  console.log(array);
});
```

## Map

```javascript
const users = [
  { id: 1, name: "John" },
  { id: 2, name: "Sara" },
  { id: 3, name: "Karen" },
  { id: 4, name: "Steve" },
];

const ids = users.map(function (user) {
  return user.id; // Return Arrays(4) [1,2,3,4]
});

console.log(ids);
```
