# Block Scope with let 1 const

```javascript
// Global Scope
var a = 1;
let b = 2;
const c = 3;

console.log("Global Scope: ", a, b, c);
// return Global Scope : 1 2 3

function test() {
  var a = 4;
  let b = 5;
  const c = 6;
  console.log("Function Scope: ", a, b, c);
  // Return Function Scope : 4 5 6
}

test(); // Only var a have the same structure

if (true) {
  //
  // Block Scope
  var a = 4;
  let b = 5;
  const c = 6;
  console.log("If Scope: ", a, b, c); // Return If Scope : 4 5 6
}
console.log("Global Scope: ", a, b, c); // Global Scope : 4 2 3 → Ce qui veut dire que b et c sont bloqué dans le bloc ce qui est plus sécurisant et évite de nombreuses erreurs Lot of programming language works like that

for (let a = 0; a < 10; a++) {
  //
  console.log(`Loop: ${a}`); // Return every new value in the scope
} //

console.log("Global Scope: ", a, b, c);
// but still return Global Scope : 1 2 3
```
